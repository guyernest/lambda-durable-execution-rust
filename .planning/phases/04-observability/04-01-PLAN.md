pub async fn agent_handler(event: AgentRequest, ctx: DurableContextHandle, llm_service: UnifiedLLMService) -> DurableResult<AgentResponse> {
    // ... config load, tool discovery, MCP connections ...
    for i in 0..max_iterations {
        let iteration_result: IterationResult = ctx.run_in_child_context(...).await?;
        let IterationResult { llm_response, assistant_message, tool_results_message, is_final } = iteration_result;
        messages.push(assistant_message);
        if is_final {
            return Ok(AgentResponse { response: llm_response });
        }
        if let Some(tool_msg) = tool_results_message { messages.push(tool_msg); }
    }
    Err(DurableError::Internal(...))
}
