// WaitConditionConfig<T> requires T: Serialize + DeserializeOwned + Send + Sync + Clone + 'static
pub struct WaitConditionConfig<T> {
    pub initial_state: T,
    pub wait_strategy: Arc<dyn Fn(&T, u32) -> WaitConditionDecision + Send + Sync>,
    pub serdes: Option<Arc<dyn Serdes<T>>>,
    pub max_attempts: Option<u32>,
}
impl<T> WaitConditionConfig<T> {
    pub fn new(initial_state: T, wait_strategy: Arc<WaitStrategy<T>>) -> Self;
    pub fn with_max_attempts(mut self, max: u32) -> Self;
}

pub enum WaitConditionDecision {
    Continue { delay: Duration },
    Stop,
}

match response {
    ToolCallResponse::Result(result) => {
        // Immediate result -- extract text content
        let text = extract_text_content(&result);
        Ok(ToolCallResult {
            tool_use_id: tool_use_id.to_string(),
            content: text,
            is_error: result.is_error,
        })
    }
    ToolCallResponse::Task(initial_task) => {
        // Long-running task -- poll via wait_for_condition (per D-07)
        // Lambda suspends between polls: zero compute cost!
        let task_id = initial_task.task_id.clone();
        let poll_ms = initial_task.poll_interval.unwrap_or(5000);

        info!(
            task_id = %task_id,
            poll_interval_ms = poll_ms,
            "Tool returned task -- polling with wait_for_condition"
        );

        // Wait strategy: stop when terminal, continue with server's poll interval
        let wait_strategy = Arc::new(move |task: &Task, _attempt: u32| {
            if task.status.is_terminal() {
                WaitConditionDecision::Stop
            } else {
                WaitConditionDecision::Continue {
                    delay: Duration::milliseconds(poll_ms as i64),
                }
            }
        });

        let config = WaitConditionConfig::new(initial_task, wait_strategy)
            .with_max_attempts(60); // ~5 min at 5s intervals

        // The check function polls tasks/get and returns updated Task state
        let client_clone = client.clone();
        let tid = task_id.clone();
        let final_task = ctx.wait_for_condition(
            Some(&format!("poll-task-{}", &task_id)),
            move |_current: Task, _step_ctx: StepContext| {
                let c = client_clone.clone();
                let id = tid.clone();
                async move {
                    c.tasks_get(&id).await
                        .map_err(|e| DurableError::Internal(e.to_string()))
                }
            },
            config,
        ).await?;

        // Retrieve the result for completed tasks
        match final_task.status {
            TaskStatus::Completed => {
                let result = client.tasks_result(&final_task.task_id).await
                    .map_err(|e| DurableError::Internal(e.to_string()))?;
                let text = extract_text_content(&result);
                Ok(ToolCallResult {
                    tool_use_id: tool_use_id.to_string(),
                    content: text,
                    is_error: result.is_error,
                })
            }
            status => {
                // Failed or Cancelled -- return as error result (LLM decides recovery)
                Ok(ToolCallResult {
                    tool_use_id: tool_use_id.to_string(),
                    content: format!("Task {} ended with status: {}", final_task.task_id, status),
                    is_error: true,
                })
            }
        }
    }
}

Before the `ToolCallResponse::Task` match arm, add a comment block:
```rust
// --- MCP Tasks: Long-Running Tool Handling ---
//
// When an MCP server needs time to process a tool call, it returns a
// CreateTaskResult (wrapped as ToolCallResponse::Task) instead of an
// immediate CallToolResult. The task has a status (Working, Completed,
// Failed, etc.) and a suggested poll_interval.
//
// Traditional approach: tokio::sleep loop (wastes Lambda compute time)
// Durable approach: ctx.wait_for_condition() suspends the Lambda between
// polls. Zero compute cost during waits. The checkpoint system remembers
// where we are, and Lambda resumes when the delay expires.
//
// This is the key insight: Durable Lambda makes MCP Tasks practically
// free to poll, even for operations that take minutes or hours.
```
