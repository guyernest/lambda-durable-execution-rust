**Step 3: Create the Durable Agent Lambda section.** Add a new section after the MCP Tester Lambda section (after the EventBridge Scheduler section). Use this exact pattern:

```typescript
// ============================================================================
// DURABLE AGENT LAMBDA - MCP Agent Execution Engine (Rust)
// ============================================================================

// Create Durable Agent Lambda (Rust) per D-01 CDK escape hatch
const durableAgentPath = path.resolve(__dirname, './functions/durable-agent-lambda/lambda-target');
const durableAgentLambda = new lambda.Function(dataStack, 'DurableAgentFunction', {
  runtime: lambda.Runtime.PROVIDED_AL2023,
  handler: 'bootstrap',
  code: lambda.Code.fromAsset(durableAgentPath),
  architecture: lambda.Architecture.ARM_64,
  timeout: Duration.minutes(15), // Agent executions can be long-running
  memorySize: 1024,              // Needs memory for LLM response processing
  environment: {
    AGENT_EXECUTIONS_TABLE: backend.data.resources.tables['AgentExecution'].tableName,
    MCP_SERVERS_TABLE: backend.data.resources.tables['McpServer'].tableName,
    LLM_MODELS_TABLE: backend.data.resources.tables['LLMModel'].tableName,
  },
});

// Grant Durable Agent Lambda permissions to read/write execution records
backend.data.resources.tables['AgentExecution'].grantReadWriteData(durableAgentLambda);

// Grant read access to config and model tables
backend.data.resources.tables['AgentConfig'].grantReadData(durableAgentLambda);
backend.data.resources.tables['LLMModel'].grantReadData(durableAgentLambda);
backend.data.resources.tables['McpServer'].grantReadData(durableAgentLambda);

// Grant permission to invoke MCP server Lambdas (agent needs to call tools)
durableAgentLambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: ['lambda:InvokeFunction'],
    resources: ['arn:aws:lambda:*:*:function:*'],
  })
);

// Grant permission to read LLM API keys from Secrets Manager (per D-15)
durableAgentLambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: ['secretsmanager:GetSecretValue'],
    resources: [
      `arn:aws:secretsmanager:${dataStack.region}:${dataStack.account}:secret:pmcp/orgs/*/agents/llm-keys*`,
    ],
  })
);

// Grant Durable Lambda checkpoint permissions (required for durable execution)
durableAgentLambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: [
      'lambda:GetDurableExecution',
      'lambda:CheckpointDurableExecution',
      'lambda:SendDurableExecutionCallbackSuccess',
      'lambda:SendDurableExecutionCallbackFailure',
    ],
    resources: ['*'],
  })
);
```

**Step 4: Wire environment variables to Amplify functions.** Add after the agent Lambda section:

```typescript
// ============================================================================
// Execute Agent Lambda Permissions
// ============================================================================

// Set environment variables for executeAgent Lambda
backend.executeAgent.addEnvironment(
  'AGENT_LAMBDA_FUNCTION_NAME',
  durableAgentLambda.functionName
);
backend.executeAgent.addEnvironment(
  'AGENT_CONFIGS_TABLE',
  backend.data.resources.tables['AgentConfig'].tableName
);
backend.executeAgent.addEnvironment(
  'AGENT_EXECUTIONS_TABLE',
  backend.data.resources.tables['AgentExecution'].tableName
);
backend.executeAgent.addEnvironment(
  'LLM_MODELS_TABLE',
  backend.data.resources.tables['LLMModel'].tableName
);
backend.executeAgent.addEnvironment(
  'MCP_SERVERS_TABLE',
  backend.data.resources.tables['McpServer'].tableName
);

// Grant executeAgent Lambda permissions to read tables
backend.data.resources.tables['AgentConfig'].grantReadData(
  backend.executeAgent.resources.lambda
);
backend.data.resources.tables['AgentExecution'].grantReadWriteData(
  backend.executeAgent.resources.lambda
);
backend.data.resources.tables['LLMModel'].grantReadData(
  backend.executeAgent.resources.lambda
);
backend.data.resources.tables['McpServer'].grantReadData(
  backend.executeAgent.resources.lambda
);

// Grant executeAgent Lambda permission to invoke the Durable Agent Lambda
backend.executeAgent.resources.lambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: ['lambda:InvokeFunction'],
    resources: [durableAgentLambda.functionArn],
  })
);

// Grant executeAgent Lambda permission to query AgentExecution GSIs
backend.executeAgent.resources.lambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: ['dynamodb:Query'],
    resources: [
      `${backend.data.resources.tables['AgentExecution'].tableArn}/index/byAgentConfigSortedByStartedAt`,
      `${backend.data.resources.tables['AgentExecution'].tableArn}/index/byOrgSortedByStartedAt`,
    ],
  })
);

// ============================================================================
// Get Execution Detail Lambda Permissions
// ============================================================================

backend.getExecutionDetail.addEnvironment(
  'AGENT_EXECUTIONS_TABLE',
  backend.data.resources.tables['AgentExecution'].tableName
);

backend.data.resources.tables['AgentExecution'].grantReadData(
  backend.getExecutionDetail.resources.lambda
);

// ============================================================================
// Manage Agent Secrets Lambda Permissions
// ============================================================================

// Grant manageAgentSecrets Lambda permissions for Secrets Manager
backend.manageAgentSecrets.resources.lambda.addToRolePolicy(
  new PolicyStatement({
    effect: Effect.ALLOW,
    actions: [
      'secretsmanager:GetSecretValue',
      'secretsmanager:CreateSecret',
      'secretsmanager:PutSecretValue',
      'secretsmanager:DeleteSecret',
    ],
    resources: [
      `arn:aws:secretsmanager:${dataStack.region}:${dataStack.account}:secret:pmcp/orgs/*/agents/llm-keys*`,
    ],
  })
);
```
