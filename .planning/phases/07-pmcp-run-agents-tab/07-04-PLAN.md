**hooks/use-agents.ts** -- Create a React hook for agent CRUD following the use-mcp-servers.ts pattern:
```typescript
'use client';

import { useState, useEffect, useCallback } from 'react';
import { generateClient } from 'aws-amplify/data';
import type { Schema } from '@/amplify/data/resource';

const client = generateClient<Schema>();

export type AgentConfig = Schema['AgentConfig']['type'];

export function useAgents() {
  const [agents, setAgents] = useState<AgentConfig[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchAgents = useCallback(async () => {
    try {
      setLoading(true);
      const { data, errors } = await client.models.AgentConfig.list();
      if (errors) {
        setError(errors.map(e => e.message).join(', '));
        return;
      }
      setAgents(data || []);
      setError(null);
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { fetchAgents(); }, [fetchAgents]);

  const createAgent = async (input: Omit<AgentConfig, 'id' | 'createdAt' | 'updatedAt'>) => {
    try {
      const { data, errors } = await client.models.AgentConfig.create({
        ...input,
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
      });
      if (errors) throw new Error(errors.map(e => e.message).join(', '));
      await fetchAgents();
      return data;
    } catch (err: any) {
      setError(err.message);
      throw err;
    }
  };

  const updateAgent = async (id: string, input: Partial<AgentConfig>) => {
    try {
      const { data, errors } = await client.models.AgentConfig.update({
        id,
        ...input,
        updatedAt: new Date().toISOString(),
      });
      if (errors) throw new Error(errors.map(e => e.message).join(', '));
      await fetchAgents();
      return data;
    } catch (err: any) {
      setError(err.message);
      throw err;
    }
  };

  const deleteAgent = async (id: string) => {
    try {
      const { errors } = await client.models.AgentConfig.delete({ id });
      if (errors) throw new Error(errors.map(e => e.message).join(', '));
      await fetchAgents();
    } catch (err: any) {
      setError(err.message);
      throw err;
    }
  };

  return { agents, loading, error, createAgent, updateAgent, deleteAgent, refetch: fetchAgents };
}

export function useAgent(id: string | null) {
  const [agent, setAgent] = useState<AgentConfig | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!id) { setLoading(false); return; }
    (async () => {
      try {
        setLoading(true);
        const { data, errors } = await client.models.AgentConfig.get({ id });
        if (errors) setError(errors.map(e => e.message).join(', '));
        else setAgent(data);
      } catch (err: any) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    })();
  }, [id]);

  return { agent, loading, error };
}
```

**hooks/use-llm-models.ts** -- Create a React hook for model registry:
```typescript
'use client';

import { useState, useEffect, useCallback } from 'react';
import { generateClient } from 'aws-amplify/data';
import type { Schema } from '@/amplify/data/resource';

const client = generateClient<Schema>();

export type LLMModel = Schema['LLMModel']['type'];

export function useLLMModels() {
  const [models, setModels] = useState<LLMModel[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchModels = useCallback(async () => {
    try {
      setLoading(true);
      const { data, errors } = await client.models.LLMModel.list();
      if (errors) {
        setError(errors.map(e => e.message).join(', '));
        return;
      }
      // Only show active models by default
      setModels((data || []).filter(m => m.isActive !== false));
      setError(null);
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { fetchModels(); }, [fetchModels]);

  return { models, loading, error, refetch: fetchModels };
}
```

```typescript
'use client';

import { useState } from 'react';
import { useRouter } from 'next/navigation';
import { LcarsHeader } from '@/components/dashboard/lcars-header';
import { AgentForm } from '@/components/agents/agent-form';
import { useAgents } from '@/hooks/use-agents';

export default function NewAgentPage() {
  const router = useRouter();
  const { createAgent } = useAgents();
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleSubmit = async (data: any) => {
    setIsSubmitting(true);

      router.push('/agents');
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className="space-y-6">
      <LcarsHeader title="Create Agent" subtitle="Configure a new AI agent" />
      <AgentForm onSubmit={handleSubmit} onCancel={() => router.push('/agents')} isSubmitting={isSubmitting} />
    </div>
  );
}
```

**app/(authenticated)/agents/[id]/edit/page.tsx** (per PMCP-02):
```typescript
'use client';

import { useState } from 'react';
import { useRouter, useParams } from 'next/navigation';
import { LcarsHeader } from '@/components/dashboard/lcars-header';
import { AgentForm } from '@/components/agents/agent-form';
import { useAgent, useAgents } from '@/hooks/use-agents';

export default function EditAgentPage() {
  const router = useRouter();
  const params = useParams();
  const agentId = params.id as string;
  const { agent, loading } = useAgent(agentId);
  const { updateAgent } = useAgents();
  const [isSubmitting, setIsSubmitting] = useState(false);

  if (loading) return <div className="animate-pulse">Loading agent...</div>;
  if (!agent) return <div>Agent not found</div>;

  const handleSubmit = async (data: any) => {
    setIsSubmitting(true);
    try {
      await updateAgent(agentId, data);
      router.push('/agents');
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className="space-y-6">
      <LcarsHeader title="Edit Agent" subtitle={agent.name || ''} />
      <AgentForm agent={agent} onSubmit={handleSubmit} onCancel={() => router.push('/agents')} isSubmitting={isSubmitting} />
    </div>
  );
}
```
