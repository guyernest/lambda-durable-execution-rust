```typescript
'use client';

import { useState, useEffect, useCallback, useRef } from 'react';
import { generateClient } from 'aws-amplify/data';
import type { Schema } from '@/amplify/data/resource';

const client = generateClient<Schema>();

export type AgentExecution = Schema['AgentExecution']['type'];

interface UseAgentExecutionsOptions {
  agentConfigId?: string;    // Filter by specific agent
  status?: string;           // Filter by status
  limit?: number;            // Page size (default 20)
}

export function useAgentExecutions(options: UseAgentExecutionsOptions = {}) {
  const [executions, setExecutions] = useState<AgentExecution[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [nextToken, setNextToken] = useState<string | null>(null);

  const fetchExecutions = useCallback(async (pageToken?: string) => {
    try {
      setLoading(true);
      // Use the auto-generated list with filters
      const listOptions: any = {
        limit: options.limit || 20,
      };
      if (pageToken) listOptions.nextToken = pageToken;

      let result;
      if (options.agentConfigId) {
        // Use the GSI via the generated query field
        result = await client.models.AgentExecution.listExecutionsByAgent(
          { agentConfigId: options.agentConfigId },
          { sortDirection: 'DESC', limit: options.limit || 20 }
        );
      } else {
        result = await client.models.AgentExecution.list(listOptions);
      }

      const { data, errors, nextToken: token } = result;
      if (errors) {
        setError(errors.map(e => e.message).join(', '));
        return;
      }

      let filtered = data || [];
      if (options.status) {
        filtered = filtered.filter(e => e.status === options.status);
      }

      if (pageToken) {
        setExecutions(prev => [...prev, ...filtered]);
      } else {
        setExecutions(filtered);
      }
      setNextToken(token || null);
      setError(null);
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, [options.agentConfigId, options.status, options.limit]);

  useEffect(() => { fetchExecutions(); }, [fetchExecutions]);

  const loadMore = () => {
    if (nextToken) fetchExecutions(nextToken);
  };

  return { executions, loading, error, nextToken, loadMore, refetch: () => fetchExecutions() };
}

// Hook for executing an agent and polling for status updates (per D-08, D-09)
export function useExecuteAgent() {
  const [executionId, setExecutionId] = useState<string | null>(null);
  const [execution, setExecution] = useState<AgentExecution | null>(null);
  const [isExecuting, setIsExecuting] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const pollingRef = useRef<NodeJS.Timeout | null>(null);

  const execute = async (agentConfigId: string, input: string) => {
    try {
      setIsExecuting(true);
      setError(null);

      // Call the custom executeAgent mutation
      const { data, errors } = await client.mutations.executeAgent({
        agentConfigId,
        input,
      });

      if (errors) throw new Error(errors.map(e => e.message).join(', '));
      if (!data) throw new Error('No response from executeAgent');

      setExecutionId(data.executionId);

      // Start polling for status updates (per D-09)
      startPolling(data.executionId);

      return data.executionId;
    } catch (err: any) {
      setError(err.message);
      setIsExecuting(false);
      throw err;
    }
  };

  const startPolling = (execId: string) => {
    // Poll every 2 seconds
    pollingRef.current = setInterval(async () => {
      try {
        const { data } = await client.models.AgentExecution.get({ id: execId });
        if (data) {
          setExecution(data);
          // Stop polling when execution is complete
          if (data.status === 'completed' || data.status === 'failed') {
            stopPolling();
            setIsExecuting(false);
          }
        }
      } catch (err) {
        // Ignore polling errors, will retry
      }
    }, 2000);
  };

  const stopPolling = () => {
    if (pollingRef.current) {
      clearInterval(pollingRef.current);
      pollingRef.current = null;
    }
  };

  // Cleanup on unmount
  useEffect(() => {
    return () => stopPolling();
  }, []);

  return { execute, executionId, execution, isExecuting, error, stopPolling };
}
```

NOTE: The `client.mutations.executeAgent` call pattern depends on the Amplify codegen. If the generated client uses a different pattern (like `client.queries.executeAgent`), the executor should adapt accordingly. The key is to call the `executeAgent` custom mutation defined in resource.ts.

import { AgentExecutionPanel } from '@/components/agents/agent-execution-panel';
