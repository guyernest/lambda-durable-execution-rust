**hooks/use-agent-metrics.ts** (per PMCP-11, PMCP-12):
Create a hook that computes metrics from AgentExecution data:
```typescript
'use client';

import { useMemo } from 'react';
import type { AgentExecution } from './use-agent-executions';
import type { LLMModel } from './use-llm-models';

interface AgentMetrics {
  totalExecutions: number;
  successRate: number;           // 0-100 percentage
  totalInputTokens: number;
  totalOutputTokens: number;
  totalCost: number;             // USD
  avgDurationMs: number;
  byAgent: Array<{
    agentName: string;
    executions: number;
    successRate: number;
    totalTokens: number;
    totalCost: number;
  }>;
  byModel: Array<{
    modelDisplayName: string;
    executions: number;
    totalInputTokens: number;
    totalOutputTokens: number;
    totalCost: number;
  }>;
  // Time series data for charts (last 30 days, grouped by day)
  dailyExecutions: Array<{
    date: string;  // YYYY-MM-DD
    total: number;
    successful: number;
    failed: number;
    tokenCount: number;
  }>;
}

export function useAgentMetrics(
  executions: AgentExecution[],
  models: LLMModel[]
): AgentMetrics {
  return useMemo(() => {
    const modelMap = new Map(models.map(m => [m.id, m]));

    const completed = executions.filter(e => e.status === 'completed');
    const failed = executions.filter(e => e.status === 'failed');

    // Calculate cost per execution
    const calculateCost = (exec: AgentExecution): number => {
      const model = modelMap.get(exec.modelId || '');
      if (!model) return 0;
      const inputCost = ((exec.inputTokens || 0) / 1000) * (model.inputPricePer1k || 0);
      const outputCost = ((exec.outputTokens || 0) / 1000) * (model.outputPricePer1k || 0);
      return inputCost + outputCost;
    };

    // Aggregate by agent
    const agentMap = new Map<string, { name: string; execs: AgentExecution[] }>();
    for (const exec of executions) {
      const key = exec.agentConfigId || 'unknown';
      if (!agentMap.has(key)) agentMap.set(key, { name: exec.agentName || 'Unknown', execs: [] });
      agentMap.get(key)!.execs.push(exec);
    }

    // Aggregate by model
    const modelAggMap = new Map<string, { name: string; execs: AgentExecution[] }>();
    for (const exec of executions) {
      const key = exec.modelId || 'unknown';
      if (!modelAggMap.has(key)) modelAggMap.set(key, { name: exec.modelDisplayName || 'Unknown', execs: [] });
      modelAggMap.get(key)!.execs.push(exec);
    }

    // Daily aggregation (last 30 days)
    const thirtyDaysAgo = new Date();
    thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
    const dailyMap = new Map<string, { total: number; successful: number; failed: number; tokenCount: number }>();
    for (const exec of executions) {
      if (!exec.startedAt) continue;
      const date = new Date(exec.startedAt);
      if (date < thirtyDaysAgo) continue;
      const key = date.toISOString().slice(0, 10);
      if (!dailyMap.has(key)) dailyMap.set(key, { total: 0, successful: 0, failed: 0, tokenCount: 0 });
      const day = dailyMap.get(key)!;
      day.total++;
      if (exec.status === 'completed') day.successful++;
      if (exec.status === 'failed') day.failed++;
      day.tokenCount += exec.totalTokens || 0;
    }

    const totalCost = executions.reduce((sum, e) => sum + calculateCost(e), 0);

    return {
      totalExecutions: executions.length,
      successRate: executions.length > 0 ? (completed.length / (completed.length + failed.length)) * 100 : 0,
      totalInputTokens: executions.reduce((sum, e) => sum + (e.inputTokens || 0), 0),
      totalOutputTokens: executions.reduce((sum, e) => sum + (e.outputTokens || 0), 0),
      totalCost,
      avgDurationMs: executions.length > 0 ? executions.reduce((sum, e) => sum + (e.durationMs || 0), 0) / executions.length : 0,
      byAgent: Array.from(agentMap.values()).map(({ name, execs }) => ({
        agentName: name,
        executions: execs.length,
        successRate: execs.length > 0 ? (execs.filter(e => e.status === 'completed').length / execs.length) * 100 : 0,
        totalTokens: execs.reduce((s, e) => s + (e.totalTokens || 0), 0),
        totalCost: execs.reduce((s, e) => s + calculateCost(e), 0),
      })),
      byModel: Array.from(modelAggMap.values()).map(({ name, execs }) => ({
        modelDisplayName: name,
        executions: execs.length,
        totalInputTokens: execs.reduce((s, e) => s + (e.inputTokens || 0), 0),
        totalOutputTokens: execs.reduce((s, e) => s + (e.outputTokens || 0), 0),
        totalCost: execs.reduce((s, e) => s + calculateCost(e), 0),
      })),
      dailyExecutions: Array.from(dailyMap.entries())
        .map(([date, data]) => ({ date, ...data }))
        .sort((a, b) => a.date.localeCompare(b.date)),
    };
  }, [executions, models]);
}
```
